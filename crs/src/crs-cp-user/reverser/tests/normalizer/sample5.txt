
INPUT ::= COMMAND_CNT { size: 4 }
          COMMAND[COMMAND_CNT]

COMMAND ::= SETUP
          | LOOKUP

SETUP ::= OPCODE { size: 4, value: 0 }
          SIZE { size: 4 }
          DATA { size: SIZE }

LOOKUP ::= OPCODE { size: 4, value: 1 }
           TABLE_SIZE { size: 4 }
           TABLE_DATA { size: TABLE_SIZE }
           SET_SIZE { size: 4 }
           SET_DATA { size: SET_SIZE }
